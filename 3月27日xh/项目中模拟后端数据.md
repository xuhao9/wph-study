#### 项目中模拟后端数据

##### 1.node原生写法

````js
const http = require('http'); // 引入http模块
const fs = require('fs'); // 读取文件模块
http.createServer((req, res) => { // 创建一个服务
    fs.readFile('./data.json', (err, data) => { // 读取
      	if (err) throw err;
        const mockData = JSON.parse(data.toString());
        res.end(JSON.stringify(mockData[req.url])); // 返回
    })
}).listen(4000, () => { // 监听端口号
    console.log('listen at 4000');
})
````







##### 2.Express写法

````js
const express = require('express');
const fs = require('fs');
const app = express();
const apiRouter = express.Router();
app.use('/', apiRouter);
apiRouter.all('*', (req, res) => {
    fs.readFile('./data.json', (err, data) => {
        if (err) throw err;
        const appData = JSON.parse(data);
        if (appData[req.path]) {
            return res.json(appData[req.path])
        }
        res.json('未找到API');
    })
})
apRouter.get('api/usr')
app.listen(3000, () => {
    console.log('listen at 3000');
})
````



##### 3.koa写法

````js
const koa = require('koa');
const fs = require('fs');

new koa().use(async (ctx) => {
    const appData = JSON.parse(fs.readFileSync('./data.json'));
    if (appData[ctx.path]) {
        ctx.body = appData[ctx.path];
    } else {
        ctx.body = '未找到API';
    }
}).listen(3000, () => {
    console.log('listen at 3000');
})


````
